-- 유저리뷰
CREATE TABLE REVIEW (
	RNO      INTEGER      NOT NULL COMMENT '게시물 번호', -- 게시물 번호
	MNO      INTEGER      NOT NULL COMMENT '회원번호', -- 회원번호
	HNO      INTEGER      NULL     COMMENT '병원번호', -- 병원번호
	TITLE    VARCHAR(255) NOT NULL COMMENT '제목', -- 제목
	CONT     TEXT         NOT NULL COMMENT '내용', -- 내용
	VCNT     INTEGER      NOT NULL COMMENT '조회수', -- 조회수
	SPART    VARCHAR(30)  NOT NULL COMMENT '시술부위', -- 시술부위
	STAT     CHAR(1)      NOT NULL COMMENT '등록상태', -- 등록상태
	HOS_NAME VARCHAR(50)  NULL     COMMENT '병원이름', -- 병원이름
	DOC_NAME VARCHAR(50)  NULL     COMMENT '담당의사', -- 담당의사
	SWAY     VARCHAR(100) NULL     COMMENT '수술방법', -- 수술방법
	SPRICE   INTEGER      NULL     COMMENT '수술가격', -- 수술가격
	SDATE    DATE         NULL     COMMENT '수술시기', -- 수술시기
	REG_DATE DATETIME     NULL     COMMENT '등록일자' -- 등록일자
)
COMMENT '유저리뷰';

-- 유저리뷰
ALTER TABLE REVIEW
	ADD CONSTRAINT PK_REVIEW -- 유저리뷰 기본키
		PRIMARY KEY (
			RNO -- 게시물 번호
		);

ALTER TABLE REVIEW
	MODIFY COLUMN RNO INTEGER NOT NULL AUTO_INCREMENT COMMENT '게시물 번호';
